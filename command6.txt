g.V().as('a').select('a');


* to assign all vertices in variable 'a' & select variable 'a', group all vertices by thier property in form of array
g.V().as('a').select('a').by('name');

g.V().as('a').properties('name').as('b').select('a', 'b').by('name').by(id);
